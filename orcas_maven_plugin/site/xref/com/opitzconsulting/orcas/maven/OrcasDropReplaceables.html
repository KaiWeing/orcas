1   package com.opitzconsulting.orcas.maven;
2   
3   import java.nio.charset.StandardCharsets;
4   import java.util.ArrayList;
5   import java.util.List;
6   
7   import org.apache.maven.plugins.annotations.Mojo;
8   import org.apache.maven.plugins.annotations.Parameter;
9   
10  import com.opitzconsulting.orcas.dbobjects.SqlplusDirAccessDbobjects;
11  
12  import de.opitzconsulting.orcas.diff.OrcasScriptRunner;
13  import de.opitzconsulting.orcas.diff.ParametersCall;
14  
15  /**
16   * Drops all replaceanles in th database-schema.
17   */
18  @Mojo( name = "dropReplaceables" )
19  public class OrcasDropReplaceables extends BaseOrcasMojo
20  {
21    /**
22     * The logname for spooling.
23     */
24    @Parameter( defaultValue = "drop-replaceables" )
25    private String logname;
26  
27    @Override
28    protected String getLogname()
29    {
30      return logname;
31    }
32  
33    /**
34     * exclude views.
35     */
36    @Parameter( defaultValue = "object_name not like '%'" )
37    private String excludewhereview;
38  
39    /**
40     * exclude object types.
41     */
42    @Parameter( defaultValue = "object_name not like '%'" )
43    private String excludewhereobjecttype;
44  
45    /**
46     * exclude packages.
47     */
48    @Parameter( defaultValue = "object_name not like '%'" )
49    private String excludewherepackage;
50  
51    /**
52     * exclude trigger.
53     */
54    @Parameter( defaultValue = "object_name not like '%'" )
55    private String excludewheretrigger;
56  
57    /**
58     * exclude functions.
59     */
60    @Parameter( defaultValue = "object_name not like '%'" )
61    private String excludewherefunction;
62  
63    /**
64     * exclude procedures.
65     */
66    @Parameter( defaultValue = "object_name not like '%'" )
67    private String excludewhereprocedure;
68  
69    @Override
70    protected void executeWithParameters( ParametersCall pParameters )
71    {
72      if( !isRunOnlyIfReplaceablesExists() || replaceablesfolder.exists() )
73      {
74        pParameters.setIsOneTimeScriptMode( false );
75        List<String> lAdditionalParameters = new ArrayList<String>();
76        pParameters.setAdditionalParameters( lAdditionalParameters );
77  
78        pParameters.setScriptUrl( SqlplusDirAccessDbobjects.getURL_delete_replacable_objects(), "delete_replacable_objects.sql", StandardCharsets.UTF_8 );
79        lAdditionalParameters.clear();
80        lAdditionalParameters.add( excludewherepackage );
81        lAdditionalParameters.add( excludewheretrigger );
82        lAdditionalParameters.add( excludewhereview );
83        lAdditionalParameters.add( excludewherefunction );
84        lAdditionalParameters.add( excludewhereprocedure );
85        new OrcasScriptRunner().mainRun( pParameters );
86  
87        pParameters.setScriptUrl( SqlplusDirAccessDbobjects.getURL_drop_all_types(), "drop_all_types.sql", StandardCharsets.UTF_8 );
88        lAdditionalParameters.clear();
89        lAdditionalParameters.add( excludewhereobjecttype );
90        new OrcasScriptRunner().mainRun( pParameters );
91      }
92      else
93      {
94        getLog().info( "no replaceables found" );
95      }
96    }
97  
98    protected boolean isRunOnlyIfReplaceablesExists()
99    {
100     return false;
101   }
102 }
